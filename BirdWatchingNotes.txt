# Replace 'domain\username' with the actual user's domain and username
$user = New-Object System.Security.Principal.NTAccount("domain\username")
$sid = $user.Translate([System.Security.Principal.SecurityIdentifier])
$sidValue = $sid.Value

$eventLogName = "Application"  # Adjust based on the actual log used by ".NET Runtime"

# Modify permissions
$sd = (wevtutil gl $eventLogName).securityDescriptor
$sd = $sd.Replace(")</sd>", "(A;;0x3;;;AU)(A;;0x3;;;IU)(A;;0x3;;;SU)(A;;0x3;;;$sidValue))</sd>")

wevtutil sl $eventLogName /sd:"$sd"
