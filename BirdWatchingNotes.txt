using Microsoft.Extensions.DependencyInjection;

public static class CorsPolicySetup
{
    public static IServiceCollection AddCustomCorsPolicies(this IServiceCollection services)
    {
        services.AddCors(options =>
        {
            options.AddPolicy("AllowSpecificOrigin", policy =>
            {
                policy.WithOrigins("http://example.com", "https://anotherdomain.com")
                      .AllowAnyHeader()
                      .AllowAnyMethod();
            });

            // Add more policies here if needed
        });

        return services;
    }
}
