$source = ".NET Runtime"
$logName = "Application"  # This is typically where such logs are written, but you can specify another if necessary

# Check if the source is registered
if (![System.Diagnostics.EventLog]::SourceExists($source)) {
    # Register the source with the Application log
    [System.Diagnostics.EventLog]::CreateEventSource($source, $logName)
    Write-Host "Source $source registered under $logName log."
} else {
    Write-Host "Source $source already exists."
    # To see which log it's currently registered to:
    $currentLog = [System.Diagnostics.EventLog]::LogNameFromSourceName($source, $machineName)
    Write-Host "Current log: $currentLog"
}
