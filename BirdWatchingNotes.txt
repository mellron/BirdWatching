using System.Text;

public class KeyManager
{
    // Method to validate if the key is 256 bits long
    public static bool IsKey256Bits(string key)
    {
        byte[] keyBytes = Encoding.ASCII.GetBytes(key);
        return keyBytes.Length == 32;  // 32 bytes = 256 bits
    }

    // Method to extend the key to 256 bits by appending zeros
    public static string ExtendKeyTo256Bits(string key)
    {
        // Check if the current key is already 256 bits
        if (IsKey256Bits(key))
        {
            return key;  // Return as is if already 256 bits
        }

        // Calculate how many zeros to add
        int zerosToAdd = 32 - Encoding.ASCII.GetBytes(key).Length;

        // Extend the key by appending zeros
        return key + new string('0', zerosToAdd);
    }
}

// Usage
string originalKey = "your_key_here";
bool isKeyValid = KeyManager.IsKey256Bits(originalKey);
Console.WriteLine("Is the original key 256 bits? " + isKeyValid);

if (!isKeyValid)
{
    string extendedKey = KeyManager.ExtendKeyTo256Bits(originalKey);
    Console.WriteLine("Extended Key: " + extendedKey);
}
