Private Sub CheckPrePaymentTypeCode(sCode As String)

    ' List of valid prepayment codes
    Dim validCodes As String() = {"BF", "YM", "YC", "ID"}
    
    ' Check if the code is valid
    Dim isValidCode As Boolean = validCodes.Contains(sCode)
    
    ' Set visibility and options based on validity of the code
    chkbox_useFHLB360Curve.Visible = isValidCode
    _bUseFHLB360CurveOption = isValidCode
    chkbox_useFHLB360Curve.Checked = False

    ' Update FHLB360Curve_Selected based on the visibility and checked state
    FHLB360Curve_Selected = (chkbox_useFHLB360Curve.Visible And chkbox_useFHLB360Curve.Checked And _bUseFHLB360CurveOption)

End Sub
