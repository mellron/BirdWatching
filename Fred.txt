public async Task MoveMultipleDirectoriesAsync(List<DirectoryInfo> directories, string destinationDirectory)
{
    // List to hold all the tasks
    List<Task> directoryMoveTasks = new List<Task>();

    // Iterate over each directory and initiate a move without awaiting
    foreach (DirectoryInfo dir in directories)
    {
        string destinationPath = Path.Combine(destinationDirectory, dir.Name);

        // Add the task to the list, but don't await it here
        directoryMoveTasks.Add(MoveDirectoryAsync(dir, new DirectoryInfo(destinationPath)));
    }

    // Await all tasks concurrently
    await Task.WhenAll(directoryMoveTasks);
}
